е®ҡж—¶еҷЁдёҺејӮжӯҘ вҖӢ

е®ҡж—¶еҷЁ вҖӢ

жҺ§еҲ¶еңЁжҹҗдёӘеӣәе®ҡзҡ„ж—¶еҲ»еҫҖиҝӣзЁӢдёӯж·»еҠ д»Јз Ғжү§иЎҢгҖӮ

setTimeoutжү§иЎҢд№ӢеҗҺиҝ”еӣһдёҖдёӘжӯЈж•ҙж•°пјҢжҳҜе®ҡж—¶еҷЁзҡ„зј–еҸ·пјҢеҸҜд»Ҙз”ЁиҝҷдёӘзј–еҸ·жё…йҷӨе®ҡж—¶еҷЁпјҢдёҚеҗҢзҡ„е®ҡж—¶еҷЁзј–еҸ·дјҡзҙҜеҠ 

  • еңЁжҹҗж®өж—¶й—ҙд№ӢеҗҺжү§иЎҢд»Јз ҒsetTimeout,

з¬¬дёҖдёӘеҸӮж•°жҳҜйңҖиҰҒжү§иЎҢзҡ„д»Јз ҒжҲ–иҖ…еӣһи°ғеҮҪж•°пјҢеҝ…йЎ»пјҢ
з¬¬дәҢдёӘеҸӮж•°жҳҜе»¶ж—¶пјҢдёҚеҶҷй»ҳи®Өдёә0пјҢеҚ•дҪҚжҜ«з§’(1s = 1000ms)
з¬¬дёүдёӘд»ҘеҗҺжҳҜйңҖиҰҒжү§иЎҢзҡ„д»Јз Ғдј е…Ҙзҡ„еҸӮж•°

js
let foo = (x,y) =>console.log(x+y);
setTimeout(foo,1000,2,3);//иҝ”еӣһдёҖдёӘжӯЈж•ҙж•°е®ҡж—¶еҷЁзј–еҸ·
//1sд№ӢеҗҺжү“еҚ°еҮә5

еӣ дёә

js
window.setTimeout === setTimeout;//true

иҝҷйҮҢдёҚз®ЎеңЁе“Әи°ғз”ЁsetTimeoutзҡ„дё»дҪ“жҳҜwindow

js
let o = {a:1}
function foo(){
    setTimeout(function(){
        console.log(this);
    },1000)
}
foo.call(o);//иҝҷйҮҢжҢҮеҗ‘window

еӣ дёәжҳҜwindowи°ғз”Ёзҡ„setTimeoutжүҖд»ҘдёҚеҸҜд»Ҙдҝ®ж”№гҖӮе®ҡж—¶еҷЁж°ёиҝңжҳҜжөҸи§ҲеҷЁи°ғз”Ё

и§ЈеҶіж–№жі•пјҡ

js
let o = {a:1}
function foo(){
    setTimeout(function(){
        console.log(this);
    }.bind(o),1000)//ејәиЎҢдҝ®ж”№еӣһи°ғеҮҪж•°зҡ„thisпјҢиҝ”еӣһзҡ„еҮҪж•°е’ҢеҺҹеҮҪж•°е·Із»ҸдёҚжҳҜеҗҢдёҖдёӘдәҶ
}
foo();//иҝҷйҮҢжҢҮеҗ‘o

жҲ–иҖ…пјҡ

js
let o = {a:1}
function foo(){
    setTimeout(()=>{//зЎ¬з»‘е®ҡеҲ°fooдёҠдёӢж–Үзҡ„thisдёӯ
        console.log(this);
    },1000)
}
foo.call(o);//иҝҷйҮҢжҢҮеҗ‘o

жё…йҷӨе®ҡж—¶еҷЁпјҡclearIntervalдј е…ҘдёҖдёӘеҸӮж•°пјҢе°ұжҳҜе®ҡж—¶еҷЁзҡ„еәҸеҸ·гҖӮйЎөйқўе”ҜдёҖгҖӮ

js
let timer = setTimeout(function(){},5000);
clearInterval(timer);//дёҠйқўйӮЈдёӘе®ҡж—¶еҷЁе°ұеҲ жҺүдәҶгҖӮ

  • иҝҗиЎҢд»Јз ҒеҗҺжҜҸйҡ”дёҖж®өж—¶й—ҙжү§иЎҢзӣёеә”д»Јз ҒпјҡsetIntervalпјҢе’ҢsetTimeoutжңүзӣёдјјжҖ§иҙЁ

    з¬¬дёҖдёӘеҸӮж•°жҳҜйңҖиҰҒжү§иЎҢзҡ„д»Јз ҒжҲ–иҖ…еӣһи°ғеҮҪж•°пјҢеҝ…йЎ»пјҢ
    з¬¬дәҢдёӘеҸӮж•°жҳҜе»¶ж—¶пјҢдёҚеҶҷй»ҳи®Өдёә0пјҢеҚ•дҪҚжҜ«з§’(1s = 1000ms)
    з¬¬дёүдёӘд»ҘеҗҺжҳҜйңҖиҰҒжү§иЎҢзҡ„д»Јз Ғдј е…Ҙзҡ„еҸӮж•°

js
let timer = setInterval(function(a,b){
    console.log(a+b);
},1000,2,3);//жҜҸйҡ”1000msжү“еҚ°дҫқж¬Ў5
clearInterval(timer);//йҖҡиҝҮе®ҡж—¶еҷЁеәҸеҸ·жё…йҷӨе®ҡж—¶еҷЁ

еҗҢжӯҘдёҺејӮжӯҘ вҖӢ

жөҸи§ҲеҷЁзҡ„jsеј•ж“ҺпјҢжҺҢжҺ§дәҶжүҖжңүзҡ„JSд»Јз ҒиҝҗиЎҢгҖӮ

JSжҳҜеҚ•зәҝзЁӢиҝҗиЎҢзҡ„пјҢжҚўеҸҘиҜқиҜҙпјҢеҗҢдёҖж—¶еҲ»еҸӘиғҪеҒҡдёҖд»¶дәӢжғ…гҖӮ

js
for(let i = 0;i < 900000000;i++){}//йЎөйқўеҚЎжӯ»дёҖе°ҸдјҡпјҢж— жі•ж“ҚдҪң

иҷҪз„¶JSжү§иЎҢе’ҢжөҸи§ҲеҷЁжёІжҹ“иҷҪз„¶дёҚжҳҜеҗҢдёҖдёӘзәҝзЁӢпјҢдҪҶжҳҜJSж¶үеҸҠйЎөйқўе…ғзҙ ж“ҚдҪңпјҢдҝ®ж”№ж“ҚдҪңе®ҢжҲҗд№ӢеүҚжёІжҹ“дјҡиў«й”Ғжӯ»пјҢJSе®ҢжҲҗд№ӢеҗҺжүҚдјҡжёІжҹ“гҖӮиҝҷе°ұжҳҜеҚ•зәҝзЁӢзҡ„жҖ§иҙЁгҖӮ

  • еҗҢжӯҘ

    еҪ“иҜ»еҸ–еҲ°JSд»Јз Ғзҡ„ж—¶еҖҷпјҡд»Јз Ғдјҡд»ҺдёҠеҫҖдёӢжү§иЎҢпјҢеҪ“еүҚйқўзҡ„д»Јз Ғжү§иЎҢе®ҢжҲҗиҝ”еӣһд№ӢеүҚпјҢеҗҺйқўзҡ„д»Јз ҒжҳҜдёҚдјҡжү§иЎҢзҡ„гҖӮе…¶дҪҷж“ҚдҪңд№ҹжҳҜдёҚдјҡжү§иЎҢзҡ„гҖӮ

js
console.log(1);
console.log(2);
for(let i = 0;i < 900000000;i++){}//йЎөйқўеҚЎжӯ»дёҖе°ҸдјҡпјҢж— жі•ж“ҚдҪң
console.log(3);//еңЁдёҠиЎҢд»Јз Ғжү§иЎҢе®ҢжҲҗд№ӢеүҚжҳҜдёҚдјҡжү“еҚ°зҡ„
//1 2 зӯүдёҖдјҡе„ҝ 3

  • ејӮжӯҘ

    ејӮжӯҘзҡ„еҗ«д№үжҳҜзӣёеҜ№дәҺеҗҢжӯҘзҡ„пјҢжҹҗиЎҢд»Јз Ғзҡ„жү§иЎҢеҚ•зӢ¬ејҖиҫҹзәҝзЁӢеӨ„зҗҶпјҢдёҺдё»зәҝзЁӢж— е…іпјҢиҝ”еӣһз»“жһңдј йҖ’еҲ°дё»зәҝзЁӢ

    дҪҶжҳҜJSиҝҗиЎҢжҳҜеҚ•зәҝзЁӢзҡ„пјҢжүҖд»ҘдёҚеӯҳеңЁзңҹжӯЈзҡ„ејӮжӯҘ

JSжү§иЎҢзҡ„ж—¶еҖҷдјҡдҫқз…§дёҖдёӘйЎәеәҸжү§иЎҢ:

е…Ҳжү§иЎҢе®ҢеҗҢжӯҘйҳҹеҲ—----->еҶҚжү§иЎҢејӮжӯҘйҳҹеҲ—

жөҸи§ҲеҷЁжү§иЎҢеҗҢжӯҘйҳҹеҲ—йҮҢзҡ„д»Јз ҒпјҢ

еҪ“йҒҮеҲ°setTimeoutжҲ–иҖ…setIntervalжҲ–иҖ…зӮ№еҮ»дәӢд»¶зӯүпјҢдјҡеңЁеҝ…иҰҒзҡ„ж—¶еҲ»е°Ҷд»Јз Ғж ҮиҜҶжү”еҲ°ејӮжӯҘйҳҹеҲ—дёӯпјҢејӮжӯҘд»Јз Ғдјҡиҝ”еӣһз»“жһңиЎЁзӨәд»Јз Ғжү§иЎҢе®ҢжҲҗпјҢе®һйҷ…д»Јз ҒжІЎжңүжү§иЎҢгҖӮ

еҗҢжӯҘжү§иЎҢе®ҢжҲҗдјҡеӨ„дәҺз©әй—ІзҠ¶жҖҒгҖӮ

еӨ„дәҺз©әй—ІзҠ¶жҖҒпјҲжү§иЎҢй—ҙйҡҷпјүжүҚдјҡдёҚеҒңжү«жҸҸејӮжӯҘйҳҹеҲ—пјҢе°Ҷжү«жҸҸеҲ°зҡ„ж ҮиҜҶдёәйңҖиҰҒжү§иЎҢзҡ„ејӮжӯҘд»Јз Ғжү§иЎҢ

еӣһи°ғ вҖӢ

жөҸи§ҲеҷЁdomдәӢд»¶пјҡе“Қеә”з”ЁжҲ·зӮ№еҮ»пјҢеӣҫзүҮеҠ иҪҪе®ҢжҲҗпјҢvueз»„д»¶зҡ„з”ҹе‘Ҫе‘Ёжңҹй’©еӯҗ д№ҹжҳҜеұһдәҺејӮжӯҘ

иҠӮжөҒе’ҢйҳІжҠ– вҖӢ

йҳІжҠ–debounce вҖӢ

еңЁдәӢд»¶иў«и§ҰеҸ‘nз§’еҗҺеҶҚжү§иЎҢеӣһи°ғеҮҪж•°пјҢеҰӮжһңеңЁиҝҷnз§’еҶ…еҸҲиў«и§ҰеҸ‘пјҢеҲҷйҮҚж–°и®Ўж—¶

иҠӮжөҒthrottle вҖӢ

и§„е®ҡдёҖдёӘеҚ•дҪҚж—¶й—ҙпјҢеңЁиҝҷдёӘеҚ•дҪҚж—¶й—ҙеҶ…пјҢеҸӘиғҪжңүдёҖж¬Ўи§ҰеҸ‘дәӢд»¶зҡ„еӣһи°ғеҮҪж•°жү§иЎҢпјҢеҰӮжһңеңЁеҗҢдёҖдёӘеҚ•дҪҚж—¶й—ҙеҶ…жҹҗдәӢд»¶иў«и§ҰеҸ‘еӨҡж¬ЎпјҢеҸӘжңүдёҖж¬ЎиғҪз”ҹж•ҲгҖӮ

жҺЁиҚҗдҪҝз”ЁLodashеҮҪж•°еә“жқҘдҪҝз”ЁиҠӮжөҒе’ҢйҳІжҠ–

иҜ·жұӮеҠЁз”»её§ вҖӢ

requestAnimationFrame(foo(time))

з”өи„‘дёҚеҚЎзҡ„ж—¶еҖҷзӣёеҪ“дәҺsetTimeout(foo,1000/60)

еӣһи°ғеҮҪж•°зҡ„еҸӮж•°timeеҸҜд»ҘиҺ·еҸ–жү“ејҖжөҸи§ҲеҷЁеҲ°зҺ°еңЁз»ҸеҺҶзҡ„жҜ«з§’ж•°

иҝҗеҠЁжһ¶ вҖӢ

CSS3еҪ“дёӯеҸҜд»ҘйҖҡиҝҮanimationеҒҡеҠЁз”»гҖӮJSдёӯзҡ„еҠЁз”»дҫқиө–иҝҗеҠЁжЎҶжһ¶гҖӮ

е·ІзҹҘиө·е§ӢдҪҚзҪ®е’Ңзӣ®ж ҮдҪҚзҪ®пјҡдј е…ҘиҝҗеҠЁж—¶й—ҙпјҢйҖҡиҝҮе®ҡж—¶еҷЁи®ҫзҪ®е®ҢжҲҗеҠЁз”»ж•ҲжһңгҖӮ

йңҖжұӮпјҡдёҖдёӘеҮҪж•°дј е…ҘеҜ№иұЎпјҢж—¶й—ҙпјҢеҸҳжҚўеҖјпјҢеӣһи°ғ(еҠЁз”»жү§иЎҢе®ҢжҲҗеҗҺжү§иЎҢзҡ„ж“ҚдҪң)

js
function getStyle(ele) {
    return ele.currentStyle || getComputedStyle(ele)
}

//з®ҖеҚ•зҡ„и§ЈеҶіе…је®№й—®йўҳ
window.requestAnimationFrame = window.requestAnimationFrame || function (cb) { return setTimeout(cb, 1000 / 60) }
window.cancelAnimationFrame = window.cancelAnimationFrame || clearTimeout

function animation(ele, data = {}, time = 500, cb) {
    /*
     * ele => е·ІиҺ·еҸ–зҡ„HTMLж ҮзӯҫеҜ№иұЎ
     * data => йңҖиҰҒж”№еҸҳзҡ„ж ·ејҸ:зӣ®ж Үж•°еҖј еҸҜд»ҘеёҰеҚ•дҪҚ
     * time => ж”№еҸҳж ·ејҸжүҖйңҖиҰҒзҡ„жҖ»ж—¶й—ҙ еҚ•дҪҚжҳҜжҜ«з§’
     * дҫӢеҰӮ : animation(box,{width:'1000px',height:'1000px'},5000)
     */
    var startValue = {}
    var changeValue = {}
    var startTime = new Date()
    var eleStart = getStyle(ele)
    for (var key in data) {
        startValue[key] = isNaN(Number.parseFloat(eleStart[key])) ? 0 : Number.parseFloat(eleStart[key])
        changeValue[key] = Number.parseFloat(data[key]) - startValue[key]
    }
    run()
    function run() {
        var nowTime = new Date() - startTime
        var t1 = nowTime / time
        for (var key in changeValue) {
            var val = t1 * changeValue[key] + startValue[key]
            ele.style[key] = val + 'px'
        }
        if (t1 >= 1) {
            cancelAnimationFrame(run)
            cb && cb()
        } else {
            requestAnimationFrame(run)
        }
    }
}
